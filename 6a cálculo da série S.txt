using System;
					
public class Program
{
	public static void Main()
	{
		 	
			
			double i=1;
			double j=1;
			double div=0;
			
			Console.WriteLine("Calculo da série: 1+3/2+5/3+ . . . +99/50");
			
			do 
			{
				i=i+2;
				j++;
				div=div+i/j;
				Console.WriteLine("i: "+i);
		        Console.WriteLine("j: "+j);
				Console.WriteLine("div: "+div);
			}
			while (j<=49);
		Console.WriteLine("A soma da série será:  "+(1+div));
		
	}
}
